    def partial_path(record_or_class, controller_path = nil)
      name = model_name_from_record_or_class(record_or_class)

      if controller_path && controller_path.include?("/")
        "#{File.dirname(controller_path)}/#{name.partial_path}"
      else
        name.partial_path
      end
    end
