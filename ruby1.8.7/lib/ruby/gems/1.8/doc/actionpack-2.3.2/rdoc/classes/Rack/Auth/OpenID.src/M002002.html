      def confirm_post_params(oid, realm, return_to, immediate)
        Rack::Response.new.finish do |r|
          r.write '<html><head><title>Confirm...</title></head><body>'
          r.write oid.form_markup(realm, return_to, immediate)
          r.write '</body></html>'
        end
      end
