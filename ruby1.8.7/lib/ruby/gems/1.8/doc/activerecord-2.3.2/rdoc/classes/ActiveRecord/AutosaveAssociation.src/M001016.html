    def self.included(base)
      base.class_eval do
        base.extend(ClassMethods)
        alias_method_chain :reload, :autosave_associations

        ASSOCIATION_TYPES.each do |type|
          base.send("valid_keys_for_#{type}_association") << :autosave
        end
      end
    end
