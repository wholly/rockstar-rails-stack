    def aggregate_context( context )
      ptr = API.sqlite3_aggregate_context( context, 4 )
      ptr.free = nil
      obj = ( ptr ? ptr.to_object : nil )
      if obj.nil?
        obj = Hash.new
        ptr.set_object obj
      end
      obj
    end
