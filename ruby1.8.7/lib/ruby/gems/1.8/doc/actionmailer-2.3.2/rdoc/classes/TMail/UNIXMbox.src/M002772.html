    def UNIXMbox.fromaddr(port)
      h = HeaderField.new_from_port(port, 'Return-Path') ||
      HeaderField.new_from_port(port, 'From') ||
      HeaderField.new_from_port(port, 'EnvelopeSender') or return 'nobody'
      a = h.addrs[0] or return 'nobody'
      a.spec
    end
