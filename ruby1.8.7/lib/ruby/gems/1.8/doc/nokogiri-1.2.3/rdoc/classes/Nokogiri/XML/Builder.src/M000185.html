      def initialize &block
        namespace = self.class.name.split('::')
        namespace[-1] = 'Document'
        @doc = eval(namespace.join('::')).new
        @parent = @doc
        @context = nil
        if block_given?
          @context = eval('self', block.binding)
        end
        instance_eval(&block) if block_given?
        @parent = @doc
      end
