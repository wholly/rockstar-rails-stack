        def self.xpath_namespace_helper rule
          rule.split(/\//).collect do |tag|
            if match = tag.match(/^(\w+:\w+)(.*)/)
              "*[name()='#{match[1]}']#{match[2]}"
            else
              tag
            end
          end.join("/")
        end
