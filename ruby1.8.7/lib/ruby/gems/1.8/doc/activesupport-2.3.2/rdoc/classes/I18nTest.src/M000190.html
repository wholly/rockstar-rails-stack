  def test_translate_with_options_using_scope_works
    I18n.backend.expects(:translate).with('de', :precision, :scope => "currency.format""currency.format")
    I18n.with_options :locale => 'de', :scope => 'currency.format''currency.format' do |locale|
      locale.t :precision
    end
  end
