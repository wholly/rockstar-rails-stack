  def test_store_translations_deep_merges_translations
    @backend.store_translations 'en''en', :foo => {:bar => 'bar'}
    @backend.store_translations 'en''en', :foo => {:baz => 'baz'}
    assert_equal Hash['en''en', {:foo => {:bar => 'bar', :baz => 'baz'}}], backend_get_translations
  end
