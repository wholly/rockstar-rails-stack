    def be_close(expected, delta)
      simple_matcher do |actual, matcher|
        matcher.failure_message = "expected #{expected} +/- (< #{delta}), got #{actual}"
        matcher.description = "be close to #{expected} (within +- #{delta})"
        (actual - expected).abs < delta
      end
    end
